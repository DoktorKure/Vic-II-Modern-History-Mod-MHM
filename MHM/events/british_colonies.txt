		year = 1964

				owner = {
					NOT = { exists = BLZ }
				}
			}
			owner = {
				release_vassal = BLZ

				diplomatic_influence = { who = BLZ value = 200 }

					government = colonial_government

						factor = 0.9
						value = conservative
						} 
					}

				}
			}
		}
		random_owned = {
			limit = {
				owner = {
					exists = BLZ
					is_our_vassal = BLZ
				}
			}
			owner = {
				BLZ = { 
					government = mandate_government 
					country_event = 800054 
					ruling_party_ideology = conservative 

					any_pop = { 
					ideology = {
						factor = 0.9
						value = conservative
						} 
					}

				}
			}
		}

		ai_chance = { factor = 0.7 }

	}

	option = {
		name = "Cede the territory to Guatemala"
		prestige = -10
		relation = { who = USA value = 50 }
		relation = { who = GUA value = 200 }
		random_owned = {
			limit = {
				owner = {
					NOT = { exists = BLZ }
				}
			}
			owner = {
				2190 = { secede_province = GUA }
			}
		}
		random_owned = {
			limit = {
				owner = {
					exists = BLZ
					is_our_vassal = BLZ
				}
			}
			owner = {
				release_vassal = BLZ
				GUA = { inherit = BLZ }
			}
		}
		
		2190 = { add_province_modifier = { name = nationalist_agitation duration = 3650 } }
		ai_chance = { factor = 0.1 }
	}

	option = {
		name = "Hold on to the lands!"
		prestige = -25
		badboy = 5
		random_owned = {
			limit = {
				owner = {
					exists = BLZ
					BLZ = { vassal_of = THIS }
				}
			}
			owner = {
				BLZ = {
					any_pop = {
						militancy = 5
						consciousness = 5
						ideology = { factor = 0.25 value = nationalist }
					}
					add_country_modifier = { name = revolution_chaos duration = 3650 } 
				}
			}
		}
		random_owned = {
			limit = {
				owner = {
					NOT = { exists = BLZ }
				}
			}
			owner = {
				any_owned = { 
					limit = { is_core = BLZ }
