#The Independence of Burma
country_event = {
	id = 1214002
	title = "The Independence of Burma"
	desc = EVTDESC1214002
	picture = "independence_of_burma2"

	fire_only_once = yes
	
	trigger = {
		tag = ENG
		year = 1948
		NOT = { has_global_flag = 1950_start_date_flag }
		NOT = { year = 1990 }
		
		government = hms_government
		war = no
		is_vassal = no
		
		#owns BUR
		OR = {
			is_possible_vassal = BUR
			is_our_vassal = BUR
		}
		
		#India is independent
		OR = {
			RAJ = { exists = yes is_vassal = no }
			HND = { exists = yes is_vassal = no }
		}
	}

	option = {
		name = "Release Burma as a republic!"
		random_owned = {
			limit = {
				owner = {
					NOT = { exists = BUR }
				}
			}
			owner = {
				release = BUR
			}
		}
		random_owned = {
			limit = {
				owner = {
					exists = BUR
					is_our_vassal = BUR
				}
			}
			owner = {
				release_vassal = BUR
			}
		}
		
		BUR = { 
			government = democracy1
			country_event = 800054
			ruling_party_ideology = socialist
			
			#Add socialist support
			any_pop = {
				limit = {
					culture = burmese
				}
				ideology = { factor = 0.25 value = socialist }
				militancy = -5
				consciousness = -5
			}
		}
		diplomatic_influence = { who = BUR value = 200 }
		
		ai_chance = { factor = 0.7 }
	}
	
	option = {
		name = "Burma's future is that of a British Dominion"
		
		prestige = -50
		badboy = 2.5
		
		random_owned = {
			limit = {
				owner = {
					NOT = { exists = BUR }
				}
			}
			owner = {
				release_vassal = BUR
				BUR = { 
					government = colonial_government
					country_event = 800054
					ruling_party_ideology = conservative 
					
					#Add nationalism
					any_pop = {
						limit = {
							culture = burmese
						}
						ideology = { factor = 0.125 value = nationalist }
						militancy = 5
						consciousness = 5
					}
					diplomatic_influence = { who = BUR value = 200 }
				}
			}
		}
		random_owned = {
			limit = {
				owner = {
					exists = BUR
					is_our_vassal = BUR
				}
			}
			owner = {
				BUR = {
					government = mandate_government
					country_event = 800054
					ruling_party_ideology = conservative 
					
					#Add nationalism
					any_pop = {
						limit = {
							culture = burmese
						}
						ideology = { factor = 0.125 value = nationalist }
						militancy = 5
						consciousness = 5
					}
				}
			}
		}
		
		ai_chance = { factor = 0.3 }
	}
	
	option = {
		name = "Burma shall remain a crown colony"

		prestige = -100
		badboy = 5

		#Add nationalism
		any_owned = { 
			limit = { is_core = BUR } 
			add_province_modifier = { name = nationalist_agitation duration = 3650 } 
		}

		any_pop = {
			limit = {
				culture = burmese
			}
			ideology = { factor = 0.25 value = nationalist }
			militancy = 10
			consciousness = 10
		}
		
		ai_chance = { factor = 0.1 }
	}

}
