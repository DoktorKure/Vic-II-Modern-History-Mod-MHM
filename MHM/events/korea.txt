  desc = EVTDESC8011905

	fire_only_once = yes

	trigger = {
    	tag = NKO
		year = 1994
		
		OR = { government = proletarian_dictatorship government = proletarian_dictatorship1 }
		war = no
		
		#Fully planned economy
		ruling_party_ideology = communist
		
		#USSR collapsed
		RUS = { NOT = { OR = { government = proletarian_dictatorship government = proletarian_dictatorship1 } } } 
		
		#Has corruption
		NOT = { has_country_modifier = minimal_government_corruption }
	}

	mean_time_to_happen = { months = 6 }

	option = {
		name = "Re-direct all food to the Leader's palace!"
		
		#Famine all over North Korea
		any_owned = {
			limit = { is_core = NKO }
			add_province_modifier = { name = severe_homeland_drought duration = 1460 }
			any_pop = { reduce_pop = 0.90 }
		}
		
		any_pop = { 
			consciousness = 5.0 
			militancy = 5 
		}
		
		ai_chance = { factor = 0.9 }
	}

}
