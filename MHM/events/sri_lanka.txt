#Independence of Ceylon
country_event = {
  id = 8028320
  title = "The Independence of Ceylon"
  desc = EVT_8028320_DESC
  picture = "ceylon_independence"
  
	fire_only_once = yes

	trigger = {
		tag = ENG
		year = 1948 
		NOT = { year = 1990 }
		
		government = hms_government
		is_vassal = no
		war = no
		
		OR = {
			is_possible_vassal = SRL
			is_our_vassal = SRL
		}
		
		#India is independent
		HND = { 
			exists = yes 
			is_vassal = no
		}
	}

	mean_time_to_happen = { months = 2 }

	option = {
		name = "Set them free!"
		prestige = 25
		random_owned = {
			limit = {
				owner = {
					exists = SRL
					SRL = { vassal_of = THIS }
				}
			}
			owner = { release_vassal = SRL }
		}
		random_owned = {
			limit = {
				owner = {
					NOT = { exists = SRL }
				}
			}
			owner = { release = SRL }
		}
		SRL = { 
			set_country_flag = full_independance
			set_country_flag = former_colonial_nation
			
			government = hms_government
			country_event = 800054 
			ruling_party_ideology = conservative
			
			any_pop = { 
				ideology = {
					factor = 0.10
					value = conservative
				} 
			}
		}
		relation = { who = SRL value = 200 }
		diplomatic_influence = { who = SRL value = 200 }
		ai_chance = { factor = 0.7 }
	}

	option = {
		name = "Delay independence"
		prestige = -50
		badboy = 2
		random_owned = {
			limit = {
				owner = {
					NOT = { exists = SRL }
				}
			}
			owner = {
				release_vassal = SRL
				SRL = {
					government = colonial_government
					country_event = 800054
					ruling_party_ideology = big_tent
				}
			}
		}
		SRL = { add_country_modifier = { name = revolution_chaos duration = 3650 } }
		diplomatic_influence = { who = SRL value = 200 }
		ai_chance = { factor = 0.2 }
	}
  
  	option = {
		name = "Hand Ceylon to India"
		relation = { who = HND value = 200 }
		random_owned = {
			limit = {
				owner = {
					exists = SRL
					SRL = { vassal_of = THIS }
				}
			}
			owner = { HND = { inherit = SRL } }
			SRL = { any_core = { add_core = HND } }
		}
		random_owned = {
			limit = {
				owner = {
					NOT = { exists = SRL }
				}
			}
			owner = {
				any_owned = { 
					limit = { is_core = SRL }
					add_core = HND
					secede_province = HND 
				}
			}
		} 
		ai_chance = { factor = 0.1 }
	}

	option = {
		name = "Ceylon shall remain British Ceylon!"
		prestige = -50
		badboy = 3
		random_owned = {
			limit = {
				owner = {
					exists = SRL
					SRL = { vassal_of = THIS }
				}
			}
			owner = {
				SRL = {
					any_pop = {
						militancy = 10
						consciousness = 10
						ideology = { factor = 0.25 value = nationalist }
					}
					add_country_modifier = { name = revolution_chaos duration = 3650 } 
				}
			}
		}
		random_owned = {
			limit = {
				owner = {
					NOT = { exists = SRL }
				}
			}
			owner = {
				any_owned = { 
					limit = { is_core = SRL } 
					any_pop = { 
						militancy = 10
						consciousness = 10
						ideology = { factor = 0.15 value = nationalist }
					} 
					add_province_modifier = { name = nationalist_agitation duration = 3650 } 
				}
			}
		}
		ai_chance = { factor = 0 }
	}
}

#Sri Lankan Civil War
country_event = {
  id = 8028321
  title = "Sri Lankan Civil War"
  desc = EVT_8028321_DESC
  picture = "srl_civil_war"
  fire_only_once = yes

	trigger = {
		tag = SRL
		year = 1983 
		NOT = { year = 1990 }
		
		is_vassal = no
		war = no
		
		is_possible_vassal = TAE
		
		HND = { exists = yes }
		
		#Sinhala only policy
		NOT = { accepted_culture = tamil }
		NOT = { accepted_culture = hindi }
	}

	mean_time_to_happen = { months = 2 }

	option = {
		name = "Crush the insurgency!"
		badboy = 5
		
		#Release the Tamil Tigers
		release = TAE
		TAE = { 
			government = sectarian_dictatorship 
			country_event = 800050 
			ruling_party_ideology = nationalist 
			relation = { who = HND value = 200 }
		}
		
		war = {
			target = TAE
			attacker_goal = { casus_belli = civil_war }
			defender_goal = { casus_belli = humiliate }
			call_ally = no
		}
		
		add_country_modifier = { name = revolution_chaos duration = 3650 }
		
		relation = { who = HND value = -200 }

		ai_chance = { factor = 0.9 }
	}

	option = {
		name = "Give in to the Tamil Tigers"
		prestige_factor = -0.25
		badboy = -5
		
		release = TAE 
		TAE = { 
			government = nationalist_dictatorship 
			country_event = 800050 
			ruling_party_ideology = nationalist 
		}
		relation = { who = TAE value = -200 }
		ai_chance = { factor = 0.1 }
	}

}
