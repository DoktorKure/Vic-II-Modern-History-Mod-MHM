#Mau Mau Uprising
country_event = {
	id = 8029800
	title = EVT_8029800_NAME 
	desc = EVT_8029800_DESC
	picture = "mau_mau_uprising" 
	
	fire_only_once = yes
	
	trigger = {
		OR = { tag = ENG tag = FBU tag = IMP }
		year = 1952 
		NOT = { year = 1990 }
		
		war = no
		is_vassal = no
		
		#owns KEN
		OR = {
			is_possible_vassal = KEN
			is_our_vassal = KEN
		}
	}
	
	mean_time_to_happen = { months = 3 }
	
	option = {
		name = "Quash the rebellion!"
		badboy = 5

		KEN = {
			any_core = { 
				add_province_modifier = { name = nationalist_agitation duration = 3650 }
				any_pop = { 
					limit = { culture = kikuyu } 
					militancy = 10 
					consciousness = 10
				}
			}
		}

		ai_chance = { factor = 0.9 }
	}
	
	option = {
		name = "Give in to the Mau Mau" 
		prestige = -100
		random_owned = {
			limit = {
				owner = {
					exists = KEN
					vassal_of = THIS
				}
			}
			owner = {
				release_vassal = KEN
			}
		}
		random_owned = {
			limit = {
				owner = {
					NOT = { exists = KEN }
				}
			}
			owner = {
				release = KEN
			}
		}
		KEN = { government = sectarian_dictatorship country_event = 800050 ruling_party_ideology = nationalist }
		ai_chance = { factor = 0.1 }
	}
	
}
