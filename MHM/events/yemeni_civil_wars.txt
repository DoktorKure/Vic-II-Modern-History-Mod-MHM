		}

			target = SYE

# EVT_8018215_NAME; 2015 Yemeni Civil War
country_event = {
	id = 8018215
	title = "Yemeni Civil War of $YEAR$"
	desc = EVT_8018215_DESC
	picture = "yemeni_civil_war"
	
	fire_only_once = yes

	trigger = {
		tag = YEM
		year = 2014
		NOT = { has_global_flag = 2015_start_date_flag }
		has_global_flag = arab_spring
		NOT = { year = 2025 }
		
		OR = { government = democracy government = democracy1 government = presidential_dictatorship government = nationalist_dictatorship government = sectarian_dictatorship }
		is_vassal = no
		war = no
		
		#Not already traditionalist
		NOT = { ruling_party_ideology = traditionalist }

		#Houthi Yemen does not exist
		HOU = { exists = no }
		
		#Owns South Yemen
		is_possible_vassal = SYE
		
		#Owns Aden
		owns = 1412
	}

	mean_time_to_happen = {
		months = 3
	}
	
	option = {
		name = EVT_8018215_A
		
		set_global_flag = yemeni_civil_war
		
		#Capital moves to Aden
		capital = 1412
		
		#Release HOU
		any_owned = {
			limit = { 
				OR = { 
					province_id = 1178
					province_id = 1179
					province_id = 1180
				} 
			}
			secede_province = HOU
		}
		
		#HOU
		HOU = {
			government = sectarian_dictatorship 
			country_event = 800050
			ruling_party_ideology = traditionalist
		}
		
		#Start civil war
		war = {
			target = HOU
			attacker_goal = {
				casus_belli = civil_war
			}
			defender_goal = {
				casus_belli = civil_war
			}
		}
		
		ai_chance = { factor = 0.9 }
		
	}
	
	option = {
		name = EVT_8018215_B
		
		prestige_factor = -0.1
		
		government = sectarian_dictatorship 
		country_event = 800050
		ruling_party_ideology = traditionalist
		
		any_pop = { 
			limit = { culture = arab_gulf } 
			militancy = 5 
			consciousness = 4 
		}
		
		ai_chance = { factor = 0.1 }
	}

}
