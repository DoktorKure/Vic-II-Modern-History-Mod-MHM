# Malta Independence
country_event = {
  id = 8004200
  title = EVT_8004200_NAME
  desc = EVT_8004200_DESC
  picture = "malta_independence"
  
	fire_only_once = yes

	trigger = {
		tag = ENG
		government = hms_government
		year = 1964 
		NOT = { year = 1990 }
		
		OR = {
			is_possible_vassal = MAL
			is_our_vassal = MAL
		}
		
		is_vassal = no
	}

	mean_time_to_happen = { months = 9 }

	option = {
		name = "Grant them independence"
		prestige = 25
		random_owned = {
			limit = {
				owner = {
					NOT = { exists = MAL }
				}
			}
			owner = {
				release = MAL
			}
		}
		random_owned = {
			limit = {
				owner = {
					exists = MAL
					is_our_vassal = MAL
				}
			}
			owner = {
				release_vassal = MAL
			}
		}
		MAL = { set_country_flag = full_independance government = hms_government1 country_event = 800054 }
		ENG = { diplomatic_influence = { who = MAL value = 200 } }
		ai_chance = { factor = 0.8 }
	}

	option = {
		name = "Follow through with the plan to integrate Malta"
		
		MAL = { 
			all_core = { 
				any_pop = { 
					limit = { culture = maltese } 
					militancy = 5 
					consciousness = 6 
					ideology = { factor = 0.15 value = nationalist }
				}
				add_province_modifier = { name = nationalist_agitation duration = 3650 }
			} 
		}
		
		#All British pops become nationalist and angry with Maltese joining the UK
		any_pop = { 
			limit = { 
				OR = { 
					is_primary_culture = yes
					is_accepted_culture = yes
				}
			} 
			militancy = 2
			consciousness = 3
			ideology = { factor = 0.05 value = nationalist }
		}
		
		add_accepted_culture = maltese
		854 = { add_core = ENG }
		
		#To turn Malta into a state
		random_owned = {
			limit = {
				owner = {
					NOT = { exists = MAL }
				}
			}
			owner = {
				release = MAL
				inherit = MAL
			}
		}
		random_owned = {
			limit = {
				owner = {
					exists = MAL
					is_our_vassal = MAL
				}
			}
			owner = { #I know this looks really weird and fucked up but it's here for a reason
				release_vassal = MAL
				inherit = MAL
				release = MAL
				inherit = MAL
			}
		}
		ai_chance = { factor = 0.2 }
	}
  
}

		tag = MAL
		year = 1974 
		NOT = { year = 1990 }

	}
