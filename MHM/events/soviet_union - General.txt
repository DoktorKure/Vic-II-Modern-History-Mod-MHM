		}

		all_core = { 
			limit = { 
				OR = { 
					AND = { is_core = UKR NOT = { is_core = CRI } }

					is_core = AZB 
					is_core = ARM 
					is_core = TKR
					is_core = MOL
					is_core = UZB

					is_core = LIT

				}
			}

		}
