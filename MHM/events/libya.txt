		tag = ENG

		prestige = 50

			limit = {
				owner = {
					NOT = { OR = { exists = LBY exists = TRI } }
				}
			}
			owner = {
				#Traditionalist Arab pops are not happy
				any_pop = {
					limit = {
						OR = { is_culture_group = arabic is_culture_group = regional_arabic_1 }
						location = { is_core = LBY }
					}

						factor = 5
