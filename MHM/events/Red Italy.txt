#The End of the ICW - Catch all clean up event for RIT tag
country_event = {
	id = 8020487
	title = "The End of the Italian Civil War"
	desc = EVTDESC8020487
	picture = "ITA_Communist_Victory"
	
	major = yes
	
	fire_only_once = yes
	
	trigger = {
		#THIS is either Italy
		OR = { tag = ITA tag = RIT }
		
		#RIT was created by Italian Civil War event
		has_global_flag = italian_civil_war
		
		#THIS is not vassal and not at war
		is_vassal = no
		war = no
		
		OR = { 
			#ITA exists but not RIT
			AND = { 
				tag = ITA 
				RIT = { exists = no } 
			}
			
			#RIT exists but not ITA
			AND = { 
				tag = RIT
				ITA = { exists = no }
			}
			
			#Both RIT and ITA share the same government type and are independent
			AND = { 
				RIT = { exists = yes OR = { government = proletarian_dictatorship government = proletarian_dictatorship1 } is_vassal = no }
				ITA = { exists = yes OR = { government = proletarian_dictatorship government = proletarian_dictatorship1 } is_vassal = no } 
			}
			AND = { 
				RIT = { exists = yes OR = { government = absolute_monarchy government = hms_government government = hms_government1 } is_vassal = no }
				ITA = { exists = yes OR = { government = absolute_monarchy government = hms_government government = hms_government1 } is_vassal = no } 
			}
			AND = { 
				RIT = { exists = yes OR = { government = democracy government = democracy1 government = socialist_democracy } is_vassal = no }
				ITA = { exists = yes OR = { government = democracy government = democracy1 government = socialist_democracy } is_vassal = no } 
			}
			
			#Other tag is vassal of THIS
			is_our_vassal = ITA
			is_our_vassal = RIT
		}
		
	}

	mean_time_to_happen = { months = 1 }

	option = {
		name = "We are a united nation once again!"
		
		clr_global_flag = italian_civil_war
		
		#If Unifier is THIS and communist
		random_owned = {
			limit = { owner = { OR = { government = proletarian_dictatorship government = proletarian_dictatorship1 government = socialist_democracy } } }
			owner = {
				prestige = 250
				war_exhaustion = -75
				any_pop = {
					ideology = { factor = 0.1 value = socialist }
					ideology = { factor = 0.3 value = communist }
					militancy = -9
					consciousness = -9
				}
				set_country_flag = italian_civil_war
			}
		}
		
		#If Unifier is the other Italy
		random_country = {
			limit = { 
				OR = { tag = ITA tag = RIT }
				OR = { government = proletarian_dictatorship government = proletarian_dictatorship1 government = socialist_democracy } 
			}
			prestige = 250
			war_exhaustion = -75
			any_pop = {
				ideology = { factor = 0.1 value = socialist }
				ideology = { factor = 0.3 value = communist }
				militancy = -9
				consciousness = -9
			}
			set_country_flag = italian_civil_war
		}
		
		#Player is ITA
		random_owned = {
			limit = { 
				owner = { ITA = { ai = no } } 
			}
			owner = { inherit = RIT }
		}
		
		#Player is RIT
		random_owned = {
			limit = { 
				owner = { RIT = { ai = no } } 
			}
			owner = { 
				inherit = ITA
				change_tag = ITA 
			}
		}
		
		#Both are AI - ITA
		random_owned = {
			limit = { 
				owner = { 
					tag = ITA
					ITA = { ai = yes } 
					RIT = { ai = yes } 
				} 
			}
			owner = { ITA = { inherit = RIT } }
		}

	}

}
