#1945 Drought
country_event = {
  id = 8023008
  title = "Romania's Drought of 1945"
  desc = EVTDESC8023008
  picture = "dry_year"
  
	fire_only_once = yes

	trigger = {
		tag = ROM
		year = 1945
		NOT = { year = 1950 }
	}

	mean_time_to_happen = { months = 1 }

	option = {
		name = "Things are not looking well..."
		
		#Add famine modifiers
		random_owned = {
			limit = { is_capital = yes continent = europe }
			owner = {
				any_owned = {
					limit = {
						has_pop_type = farmers
						is_colonial = no
						continent = europe
						is_overseas = no
						NOT = {
							#Desert
							terrain = desert
							terrain = coastal_desert
							terrain = semi_desert
							terrain = semi_desert_hills
							terrain = semi_desert_mountains
							terrain = desert_hills
							terrain = desert_mountains
							terrain = cold_desert
							terrain = cold_desert_hills
							terrain = cold_desert_mountains
				
							#Tundra
							terrain = tundra
							terrain = tundra_hills
							terrain = tundra_mountains
				
							#Arctic
							terrain = arctic
						}
					}
					add_province_modifier = { name = severe_homeland_drought duration = 730 }
				}
			}
		}
		
		set_country_flag = romania_drought_1945
		
	}
}
