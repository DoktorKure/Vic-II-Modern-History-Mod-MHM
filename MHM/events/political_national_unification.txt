        }

	}

			value = -100
		}
		FROM = {
			prestige = -20

				who = THIS
				value = -50
			}
		}
		leave_alliance = FROM
		any_pop = {
			militancy = 3
			consciousness = 2
		}
		ai_chance = { factor = 0 }
	}
}

#Transfer spherelings to superstate
country_event = {
	id = 11109
	title = "Transfered to Superstate"
	desc = "As we were in the sphere of a country that has been diplomatically annexed into a superstate, we are now their spherelings."
	picture = "gain_core"

	fire_only_once = yes
	
	is_triggered_only = yes
	
	immediate = { clr_country_flag = sphere_to_superstate }

	option = {
		name = "We have no choice but to accept this reality."
		FROM = { diplomatic_influence = { who = THIS value = 500 } }
		ai_chance = {
			factor = 0.99
			modifier = {
				factor = 0
				AND = {
					FROM = { NOT = { OR = { government = nationalist_dictatorship ruling_party_ideology = nationalist ruling_party_ideology = populist } } }
					OR = { government = nationalist_dictatorship ruling_party_ideology = nationalist ruling_party_ideology = populist }
				}
			}
		}
	}

	option = {
		name = "We will go our own way."
		FROM = {
			casus_belli = {
				target = THIS
				type = add_to_sphere
				months = 12
			}
		}
		ai_chance = { factor = 0.01 }
	}	
}
